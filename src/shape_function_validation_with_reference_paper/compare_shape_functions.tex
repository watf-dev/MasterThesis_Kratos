\newlength{\figw}

% One subfigure cell; #3 = width factor (e.g. \linewidth or 1.1\linewidth)
\newcommand{\cell}[3]{%
  \begin{subfigure}[b]{\figw}
    \includegraphics[width=#3]{#1}
    % \caption*{\small #2}
  \end{subfigure}%
}

% Row of 4: two ref/mine pairs for files #1 and #2
\newcommand{\quadrow}[2]{%
  \begin{figure}[H]
    \centering
    \cell{\refdir/#1}{Ref #1}{\linewidth}\hfill
    \cell{\mydir/#1}{Mine #1}{1.1\linewidth}\hfill
    \cell{\refdir/#2}{Ref #2}{\linewidth}\hfill
    \cell{\mydir/#2}{Mine #2}{1.1\linewidth}
  \end{figure}%
}

% Row of 2: ref at col 1, mine at col 4 (empty boxes hold cols 2 and 3)
\newcommand{\pairrow}[1]{%
  \begin{figure}[H]
    \centering
    \cell{\refdir/#1}{Ref #1}{\linewidth}\hfill
    \cell{\mydir/#1}{Mine #1}{1.1\linewidth}\hfill
    \makebox[\figw]{}\hfill
    \makebox[\figw]{}
  \end{figure}%
}
